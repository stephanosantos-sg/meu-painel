function doGet(e) { var action = e && e.parameter && e.parameter.action; if (action === 'events') { var now = new Date(); var end = new Date(); end.setDate(now.getDate() + 15); var cal = CalendarApp.getDefaultCalendar(); var evs = cal.getEvents(now, end); var result = evs.map(function(ev) { return {title:ev.getTitle(),start:ev.getStartTime().toISOString(), end:ev.getEndTime().toISOString(),allDay:ev.isAllDayEvent(), location:ev.getLocation()||''}; }); return ContentService.createTextOutput( JSON.stringify(result)).setMimeType( ContentService.MimeType.JSON); } var s = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Data'); return ContentService.createTextOutput( s.getRange('A1').getValue()).setMimeType( ContentService.MimeType.JSON); } function doPost(e) { var s = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Data'); s.getRange('A1').setValue(e.postData.contents); return ContentService.createTextOutput('ok'); }
